Chapter 3 Table

I process the table named extpol_pro_s3e_vsb_sacz0_rod_terpre_rescyc_des_cov_raw. It is one of the tables that I worked with. Its title is a mix of abbreviations. nom=colnames(dt) # names of the dataset's columns
# traduction of the columns' names
for(i in seq(1:length(nom))){
  if(i!=7){ # column 7 is impossible to translate (n01)
    colnames(dt)[i] <- tra.str.F(nom[i],'namwri')
  }
}
  
# traduction of the content
for(i in seq(1:5)){
  for(j in seq(1:nrow(dt)+1)){
    dt[j,i] <- tra.str.F(toString(dt[j,i]),'namwri')
  }
}

You can find here the script used by the convert_table function :

convert_table = function(tt, col.use, name){ # col.use = liste des colonnes constantes, tt = dataframe
tt.v.list = list()
for (i in seq_along(col.use)) {
  tt.v.list [[i]] = as.vector (unlist (unique(tt[col.use[i]])))
}
tt.v.list =  (do.call (rbind, tt.v.list))
tt.v.list.name = paste(as.vector (tt.v.list),collapse =  " # ")

tt.red = tt [-col.use]
x = paste(names(tt.red)[1],tt.v.list.name, sep= " # ")
tt.red = tt.red [-1]
names(tt.red)[1] = paste(names(tt.red)[1],x , sep= " # ")
return(knitr::kable(tt.red, format = "html", booktabs = TRUE, caption = name)%>%

}
