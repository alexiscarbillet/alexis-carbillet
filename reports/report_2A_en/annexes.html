Annexes

In this part, there are different codes used during the translation of words present in the initial datasets are presented. The first one is named INI_danter, it contains the data which allowed the translation and the second one is named tra.str.F, it is the function which translate, it relies on INI-danter. Those two use the R language and are implemented in most of the codes showed before in this report. There are also the two R scripts which are used in order to plot the dynamics graphics from the template of the chapter “Dynamic report generation”.

5.2 Tra.str.F.R

The second is the tra.str.F.R :

tra.str.F = function (tra.str, grodankid) {
  tra.df.tem = as_tibble(tra.str)
  tra.df.tem$value = paste0(1:nrow(tra.df.tem), "_", tra.df.tem$value)
  tra.df.tem = split(tra.df.tem, tra.df.tem$value)
  # equivalent to seq (1: lenght (tra.df.tem))
  for (i in seq_along(tra.df.tem)) { #
    tra.df.tem[i] = strsplit(tra.df.tem[[i]]$value, '_')
  }
  # keep the last word of each
  # equivalent to seq (1: lenght (tra.df.tem))
  for (i in seq_along(tra.df.tem)) { 
    tra.df.tem[[i]] = tra.df.tem[[i]][-1]
  }
  # equivalent to seq (1: lenght (tra.df.tem))
  for (i in seq_along(tra.df.tem)) { 
    for (j in 1:length(tra.df.tem[[i]])) {
      temp = danter %>% filter(nid %in% tra.df.tem[[i]][j])
      tra.df.tem[[i]][j] = as.vector (unlist (temp[grodankid]))
    }
  }
  
  # equivalent to seq (1: lenght (tra.df.tem))
  for (i in seq_along(tra.df.tem)) { 
    tra.df.tem[[i]] = paste (tra.df.tem[[i]], collapse = " ")
  }
  
  
  names (tra.df.tem) = as.numeric (do.call (rbind, lapply (strsplit(
    names(tra.df.tem), '_'
  ), '[', 1)))
   # equivalent to seq (1: lenght (tra.df.tem))
  for (i in seq_along(tra.df.tem)) {
    tra.df.tem[[i]] = paste (names (tra.df.tem[i]), tra.df.tem[[i]], sep = "_")
  }
  
  tra.df.tem = as.tibble (do.call (rbind, tra.df.tem))
  tra.df.tem$order  = as.numeric (do.call (rbind, lapply (strsplit(tra.df.tem[[1]], "_"), "[", 1)))
  # rearrange tra.df.tem according to $order
  tra.df.tem = tra.df.tem [order(tra.df.tem$order),]['V1'] 
  tra.df.tem = as.vector (unlist (tra.df.tem))
  # replaces the first match of all elements.
  tra.df.tem = sub('.*_', '', tra.df.tem) 
  
  rownames(tra.df.tem) = NULL
  
  if (length(tra.str) == length(tra.df.tem)) {
    return(tra.df.tem)
  } else {
    # returns the product set of elements tra.str and tra.df.tem
    return (cross2(tra.str, tra.df.tem)) 
  }
}

5.3 Works by this artist

library(dplyr)
json <- jsonlite::fromJSON("watercolors.json")
watercolors <- as.data.frame(json)
watercolors$year <- as.integer(watercolors$year)
watercolors <- watercolors[!is.na(watercolors$year),]
allFreq <- plyr::count(na.omit(as.integer(watercolors$year)))
# ignore the line where a data is missing
names(allFreq) <- c("year", "collection")
artistFreq <- plyr::count(na.omit(as.integer("1908")))
names(artistFreq) <- c("year", "artist")
mfreq <- merge(allFreq, artistFreq, all = TRUE)
mfreq[is.na(mfreq)] <- 0
# make sure we show at least a 10 years span
window <- c(max(min(artistFreq$year) - 5, min(mfreq$year)),
            min(max(artistFreq$year) + 5, max(mfreq$year)))
dygraphs::dygraph(mfreq, main = "works by this artist") %>% 
  dygraphs::dyAxis("y", axisLabelWidth = 30) %>%
  dygraphs::dyOptions(drawPoints = TRUE, pointSize = 2) %>%
  dygraphs::dyLegend(width = 300) %>%
  dygraphs::dyRangeSelector(dateWindow = window)
