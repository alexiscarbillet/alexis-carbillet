import matplotlib.pyplot as plt

import numpy as np

import random

import math


def poisson (mu,k): # amplified by a 10 coefficient

return 10*mu**k*np.exp(-mu)/math.factorial(k)

def dark_noise(n): # n: number of points

a=[]

for i in range(n):

a.append(random.random()) # create noise

return a # return list which contains the noise


def readout_noise(n,f,mu,p): # p the landing

a=[]

for i in range(n):

if(f(mu,i)>p): # photons arrive on the pixel

a.append(3*random.random()) #readout_noise

else:

a.append(0) # no noise because there are no photons arriving on these pixels

return a # returns the list containing the photon reading noise

# we trace the signal that would be sent by the camera

def trace(n,f,mu,p):

x=[]

y=[]

for i in range(n):

x.append(i) # list of abscissa, they are integers

y.append(poisson(mu,i)+p+dark_noise(n)[i]+readout_noise(n,f,mu,p)[i])

# noisy signal

plt.plot(x,y) # graph creation

plt.xlabel('x value') # abscissa axis legend

plt.ylabel('value of the fish law') # ordinate axis legend

plt.title('loi de poisson de paramètre 10 bruité') # graphic title

plt.show() # plot graphic

def trace2(n,f,mu,p): # SNR trace according to the pixel interval around the most receptive

x=[]

y=[]

bruit=[]

signal=[]

for i in range(n):

x.append(i) # list of abscissa, they are integers

y.append(poisson(mu,i)+p+dark_noise(n)[i]+readout_noise(n,f,mu,p)[i])

bruit.append(dark_noise(n)[i]+readout_noise(n,f,mu,p)[i]) # noisy signal

signal.append(poisson(mu,i))

m=max(signal) # look for the signal peak (maximum of the ordinates)

k=signal.index(m) # look for the pixel that receives the maximum photon (the most receptive)

signal_total=0 # SNR numerator

bruit_total=0 # SNR denominator

SNR=[]

rayon=[]

for r in range(1,min(n-k,k)): # look at the pixels located in the circle of radius r and the one that receives the maximum number of photons

for j in range(k-r,k+r): # look at the noise around this pixel for a given radius

signal_total+=signal[j] # value addition

bruit_total+=bruit[j] # value addition

SNR.append(signal_total/bruit_total) # divide by the corresponding noise

rayon.append(r)

plt.plot(rayon,SNR) # graphic creation

plt.xlabel('radius value') # abscissa axis legend

plt.ylabel('SNR value') # ordinate axis legend

plt.title('SNR according to the value of the radius') # graphic title

plt.show() # plot graphic

trace2(21,poisson,10,1) # executes the trace2 function with 21 points, centered on 10

def getMappaBadFromMedianAndStdFrames(self, bias, biasv, tit):

from scipy.ndimage.filters import gaussian_filter # import library

filbias=gaussian_filter(bias, 5) # Gauss filter application

mappaHot=(bias>3*np.std(filbias)+ filbias)|(bias<1.) # erroneous by heat

mappaDancer=biasv>3*np.std(biasv) + filbias # erroneous by their volatility

mappaBad= mappaHot+ mappaDancer # wrong pixels = sum of erroneous one

imshow(mappaBad, origin='lower')

title('Bad pixels fraction= %.2f%%, '%(np.sum(mappaBad)/(mappaBad.shape[0]*mappaBad.shape[1])*100)+tit) # image title

print(np.sum(mappaBad)/(mappaBad.shape[0]*mappaBad.shape[1])*100) # displays the image of the wrong pixels so their positions on the sensor
